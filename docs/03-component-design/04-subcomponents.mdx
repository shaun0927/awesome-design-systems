## React Compound Component (Context API)

```tsx title="CompoundCard.tsx"
import React, { createContext, useContext } from 'react';

// Shared context for parent-child communication
interface CardContextValue {
  variant: 'default' | 'outlined' | 'elevated';
  colorMode: 'light' | 'dark';
}

const CardContext = createContext<CardContextValue>({
  variant: 'default',
  colorMode: 'light',
});

// Parent container
function CardRoot({
  variant = 'default',
  colorMode = 'light',
  children,
}: CardContextValue & { children: React.ReactNode }) {
  return (
    <CardContext.Provider value={{ variant, colorMode }}>
      <article className={`card card--${variant} card--${colorMode}`}>
        {children}
      </article>
    </CardContext.Provider>
  );
}

// Subcomponents consume parent context
function CardMedia({ src, alt }: { src: string; alt: string }) {
  return (
    <div className="card__media">
      <img src={src} alt={alt} loading="lazy" />
    </div>
  );
}

function CardText({
  title,
  description,
}: {
  title: string;
  description?: string;
}) {
  const { colorMode } = useContext(CardContext);
  return (
    <div className={`card__text card__text--${colorMode}`}>
      <h3 className="card__title">{title}</h3>
      {description && <p className="card__desc">{description}</p>}
    </div>
  );
}

function CardActions({ children }: { children: React.ReactNode }) {
  return <div className="card__actions">{children}</div>;
}

// Compound component export (similar to Radix UI pattern)
export const Card = Object.assign(CardRoot, {
  Media: CardMedia,
  Text: CardText,
  Actions: CardActions,
});

// Usage:
// <Card variant="elevated" colorMode="dark">
//   <Card.Media src="photo.jpg" alt="Description" />
//   <Card.Text title="Product" description="Details here" />
//   <Card.Actions>
//     <Button>Buy</Button>
//     <Button variant="ghost">Save</Button>
//   </Card.Actions>
// </Card>
```
