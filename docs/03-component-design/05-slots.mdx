## Slot Implementations: React, Vue, Web Components

### React: children + render props

```tsx title="SlottedRow.tsx"
interface RowProps {
  // Named slots as props
  leadingVisual?: React.ReactNode;
  trailingActions?: React.ReactNode;
  // Default slot
  children: React.ReactNode;
}

export function Row({ leadingVisual, trailingActions, children }: RowProps) {
  return (
    <div className="row">
      {leadingVisual && <div className="row__leading">{leadingVisual}</div>}
      <div className="row__content">{children}</div>
      {trailingActions && <div className="row__trailing">{trailingActions}</div>}
    </div>
  );
}

// Usage:
// <Row
//   leadingVisual={<Avatar src="user.jpg" />}
//   trailingActions={<IconButton icon="more" />}
// >
//   <Text variant="body">Row content here</Text>
// </Row>
```

### Vue: Named Slots

```vue title="SlottedRow.vue"
<template>
  <div class="row">
    <div v-if="$slots.leading" class="row__leading">
      <slot name="leading" />
    </div>
    <div class="row__content">
      <slot />  <!-- default slot -->
    </div>
    <div v-if="$slots.trailing" class="row__trailing">
      <slot name="trailing" />
    </div>
  </div>
</template>

<!-- Usage:
<Row>
  <template #leading><Avatar src="user.jpg" /></template>
  Row content here
  <template #trailing><IconButton icon="more" /></template>
</Row>
-->
```

### Web Components: Shadow DOM Slots

```html title="slotted-row.html"
<template id="row-template">
  <style>
    :host { display: flex; align-items: center; gap: 0.75rem; }
    .leading, .trailing { flex-shrink: 0; }
    .content { flex: 1; }
  </style>
  <div class="leading"><slot name="leading"></slot></div>
  <div class="content"><slot></slot></div>
  <div class="trailing"><slot name="trailing"></slot></div>
</template>

<script>
class SlottedRow extends HTMLElement {
  constructor() {
    super();
    const shadow = this.attachShadow({ mode: 'open' });
    const template = document.getElementById('row-template');
    shadow.appendChild(template.content.cloneNode(true));
  }
}
customElements.define('slotted-row', SlottedRow);
</script>

<!-- Usage:
<slotted-row>
  <img slot="leading" src="avatar.jpg" />
  <span>Row content</span>
  <button slot="trailing">Action</button>
</slotted-row>
-->
```
