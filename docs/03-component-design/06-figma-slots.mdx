## Figma Slots to TypeScript Mapping

```typescript title="figma-to-react-mapping.ts"
// Figma slot definition (YAML spec)
// items:
//   type: slot
//   anyOf: checkbox
//   minItems: 1
//   maxItems: undefined

// Equivalent React TypeScript interface
interface CheckboxGroupProps {
  /** Slot: accepts only Checkbox children (min 1) */
  children: React.ReactElement<CheckboxProps> | React.ReactElement<CheckboxProps>[];
  label: string;
  error?: string;
}

// With runtime validation
function CheckboxGroup({ children, label, error }: CheckboxGroupProps) {
  const items = React.Children.toArray(children);

  if (items.length < 1) {
    console.warn('CheckboxGroup requires at least 1 Checkbox child');
  }

  return (
    <fieldset className="checkbox-group" role="group" aria-labelledby={`${label}-label`}>
      <legend id={`${label}-label`}>{label}</legend>
      <div className="checkbox-group__items">{children}</div>
      {error && <span className="checkbox-group__error" role="alert">{error}</span>}
    </fieldset>
  );
}

// Figma "preferred values" â†’ TypeScript generic constraint
interface SlotConfig<T extends React.ElementType> {
  component: T;
  minItems?: number;
  maxItems?: number;
}

// Example: Tab slot only accepts TabItem
type TabSlot = SlotConfig<typeof TabItem>;
```
