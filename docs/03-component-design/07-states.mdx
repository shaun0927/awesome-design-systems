## CSS State Priority Implementation

```css title="input-states.css"
/* Base styles */
.input {
  border: 2px solid var(--color-border);
  border-radius: var(--radius-md);
  padding: var(--space-s) var(--space-m);
  background: var(--color-surface);
  transition: border-color 0.2s, box-shadow 0.2s;
}

/* Interactive states (lowest priority) */
.input:hover:not(:disabled):not([readonly]) {
  border-color: var(--color-border-hover);
}
.input:focus-visible {
  border-color: var(--color-focus);
  box-shadow: 0 0 0 3px var(--color-focus-ring);
  outline: none;
}

/* Validation states (medium priority - overrides interactive) */
.input[aria-invalid="true"] {
  border-color: var(--color-error);
}
.input[aria-invalid="true"]:focus-visible {
  box-shadow: 0 0 0 3px var(--color-error-ring);
}
.input[data-validation="success"] {
  border-color: var(--color-success);
}

/* Control states (highest priority - overrides everything) */
.input[readonly] {
  background: var(--color-surface-muted);
  cursor: default;
}
.input:disabled {
  opacity: 0.5;
  cursor: not-allowed;
  border-color: var(--color-border);
  /* Disabled overrides all other visual states */
}
```

```typescript title="state-types.ts"
// Separate concerns in TypeScript
interface InteractiveState {
  state: 'rest' | 'hover' | 'active' | 'focus';
}

interface ControlState {
  disabled?: boolean;
  readOnly?: boolean;
}

interface ValidationState {
  validation?: 'none' | 'error' | 'success';
}

// Combined input props with priority rules
type TextInputProps = InteractiveState & ControlState & ValidationState & {
  value: string;
  onChange: (value: string) => void;
  label: string;
};

// State resolution function
function resolveVisualState(props: ControlState & ValidationState & InteractiveState) {
  if (props.disabled) return 'disabled';      // Priority 1
  if (props.readOnly) return 'readonly';       // Priority 2
  if (props.validation === 'error') return 'error';  // Priority 3
  if (props.validation === 'success') return 'success';
  return props.state;                          // Priority 4
}
```
