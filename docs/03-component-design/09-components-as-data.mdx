## YAML to React Code Generator

```yaml title="alert.component.yaml"
name: Alert
description: "Feedback message with severity levels"
anatomy:
  root:
    type: container
  icon:
    type: slot
    nullable: true
  content:
    type: container
  title:
    type: text
  description:
    type: text
    nullable: true
  dismiss:
    type: slot
    nullable: true

props:
  severity:
    type: string
    default: info
    enum: [info, success, warning, error]
  dismissible:
    type: boolean
    default: false
  title:
    type: string
    required: true
  description:
    type: string

default:
  elements:
    root:
      styles:
        display: flex
        alignItems: flex-start
        gap: foundations/space/s
        padding: foundations/space/m
        borderRadius: foundations/radius/md
        backgroundColor: foundations/color/surface/info
```

```typescript title="generate-component.ts"
import { readFileSync } from 'fs';
import { parse } from 'yaml';

interface ComponentSpec {
  name: string;
  anatomy: Record<string, { type: string; nullable?: boolean }>;
  props: Record<string, {
    type: string;
    default?: unknown;
    enum?: string[];
    required?: boolean;
  }>;
}

function generateReactComponent(yamlPath: string): string {
  const spec: ComponentSpec = parse(readFileSync(yamlPath, 'utf8'));

  const propsInterface = Object.entries(spec.props)
    .map(([name, def]) => {
      const tsType = def.enum
        ? def.enum.map(v => `'${v}'`).join(' | ')
        : def.type === 'boolean' ? 'boolean' : 'string';
      const optional = def.required ? '' : '?';
      return `  ${name}${optional}: ${tsType};`;
    })
    .join('\n');

  const slots = Object.entries(spec.anatomy)
    .filter(([, v]) => v.type === 'slot')
    .map(([name]) => `  ${name}?: React.ReactNode;`)
    .join('\n');

  return `
interface ${spec.name}Props {
${propsInterface}
${slots}
  children?: React.ReactNode;
}

export function ${spec.name}({ ${Object.keys(spec.props).join(', ')}, children }: ${spec.name}Props) {
  return (
    <div className="${spec.name.toLowerCase()}" role="alert">
      {children}
    </div>
  );
}`.trim();
}
```
