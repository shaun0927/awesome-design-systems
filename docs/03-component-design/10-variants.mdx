## CSS Cascade Variant Implementation

```css title="alert-variants.css"
/* Default styles (base layer) */
.alert {
  display: flex;
  align-items: flex-start;
  gap: var(--space-s);
  padding: var(--space-m);
  border-radius: var(--radius-md);
  border-left: 4px solid var(--color-info);
  background: var(--color-info-subtle);
  color: var(--color-text-primary);
}

/* Appearance variants (cascade layer 1) */
.alert[data-appearance="success"] {
  border-left-color: var(--color-success);
  background: var(--color-success-subtle);
}
.alert[data-appearance="warning"] {
  border-left-color: var(--color-warning);
  background: var(--color-warning-subtle);
}
.alert[data-appearance="critical"] {
  border-left-color: var(--color-error);
  background: var(--color-error-subtle);
}

/* Size variants (cascade layer 2 - overrides padding/font) */
.alert[data-size="small"] {
  padding: var(--space-xs) var(--space-s);
  font-size: var(--text-sm);
}
.alert[data-size="large"] {
  padding: var(--space-l);
  font-size: var(--text-lg);
}

/* Compound variants (specific combination overrides) */
.alert[data-appearance="critical"][data-size="large"] {
  border-left-width: 6px;
}
```

```typescript title="generate-stories.ts"
// Generate Storybook stories from variant YAML data
import { parse } from 'yaml';

interface VariantSpec {
  props: Record<string, { enum?: string[]; type: string; default?: unknown }>;
}

function generateStories(spec: VariantSpec): string {
  const stories: string[] = [];

  for (const [propName, propDef] of Object.entries(spec.props)) {
    if (!propDef.enum) continue;

    for (const value of propDef.enum) {
      const storyName = `${propName}_${value}`.replace(/[^a-zA-Z0-9]/g, '_');
      stories.push(`
export const ${storyName}: Story = {
  args: { ${propName}: '${value}' },
};`);
    }
  }

  return stories.join('\n');
}
```
